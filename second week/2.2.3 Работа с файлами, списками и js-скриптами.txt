#2.2.3 Работа с файлами, списками и js-скриптами

from selenium import webdriver
from selenium.webdriver.support.ui import Select
import time

link = "http://suninjuly.github.io/selects1.html"

browser = webdriver.Chrome()
browser.get(link)
import math

a = browser.find_element_by_id("num1")
a1 = a.text
b = browser.find_element_by_css_selector("form h2 span:nth-child(3)")
b1 = b.text
c = browser.find_element_by_id("num2")
c1 = c.text
print(a1,b1,c1)
d = int(a1)+int(c1)
print(d)

str_d = str(d)
print(str_d)

select = Select(browser.find_element_by_tag_name("select"))
select.select_by_visible_text(str_d) 


button = browser.find_element_by_class_name("btn")
button.click()

time.sleep(10)

browser.quit()
